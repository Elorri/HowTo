# Add a toolbar and floating action button

	<Framelayout>
		<TextView layout_marginTop="160dp"/>
		<Toolbar layout_height="160dp"/>
		<Button layout_marginTop="160dp" layout_gravity="right"/>
	</Framelayout>
	
or 

	<Framelayout>
		<Linearlayout>
			<Toolbar layout_height="160dp"/>
			<TextView />
		</Linearlayout>
		<Button layout_marginTop="160dp" layout_gravity="right"/>
	</Framelayout>
